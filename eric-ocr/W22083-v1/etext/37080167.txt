ཐབས་སུ་གླན་ནོ། །རྐྱེན་ཇི་ལྟ་བ་བཞིན་དུ་བརྒྱ་སྟེ། ཆོས་གོས་འདི་ནི་རུང་བ་སྤྱད་པར་འོས་པའོ། །དེ་བཞིན་དུ་ལན་གཉིས་ལན་གསུམ་དུ་བཟླས་། མཁན་པོས་འ<line>
ཐབས་ཡན་ནོ་ཞེས་བརྗོད་པར་བྱའོ། །སློབ་མས་ལེགས་སོ་ཞེས་བརྗོད་པར་བྱའོ། །དེའི་འོག་ཏུ་རྗོས་གོས་སླྲ་གོས་རྒྱུ་བཟུང་ནས། དེས་འདི་སྐད་ཅེས་བརྗོད་པར་བྱ་སྟེ། མཁན་པོ་འ<line>
དགོངས་སུ་གསོལ་། བདག་ས་ངེ་འདི་ཞེས་བགྱ་བོའི་ཆོས་ཀོས་འདི་བླ་གོས་སུ་བྱིུན་གྱིས་བརླབ་བོ། །འཚལ་ན་སྙམ་བྲིན་བདུན་པར་བགྱི་སྟེ། གླེགས་བུ་བྱེད་དང་གསུམ་པར་བགྱིའོ། །།<line>
བར་ཆུད་མ་བྱུང་ན་བཀྲུའོ། །བརྐྱང་ངོ་། །དྲའོ། །བསླེབ་བོ། །གཟུངས་བདབ་བོ། །དྲུབ་བོ། །ཁ་བསྒྱུར་རོ། །ཡང་ནྣ་འདི་ལ་སྤན་ཐབས་སུ་གླན་ནོ། །རྐྱེན ཇི་ལྟ་བ་བཞིན་དུ་བགྱིསྟ།།<line>
རློས་ཀོས་འདི་རུང་བ་སྙྱད་པར་འོས་པའོ། །དེ་བཞིན་དུ་ལན་གཞིས་གན་གསུམ་དུ་བཟླས། མཁན་པོས་ཐབས་ཡན་ནོ་ཞེས་བརྗོད་པར་བྱའོ། །སློབ་མས་ལེགས་སོ་ཤེས་བརྗོད་པར་ ས<line>
བྱའོ༑ ༑དེའི་འོག་ཏུ་རྗོས་ཀོས་མཐང་གོས་རྒྱུ་བ་ཟུང་ནས། དེས་འདི་སྐད་ཅེས་བརྗོད་པར་བྱ་སྟེ། མཁན་པོ་དགོངས་སུ་གསོལ། བདག་མིང་འདི་ཞེས་བགྱི་བའི་རྗོས་ཀོས་འདི་ ས<line>
མཐང་གོས་སུ་བྱིན་གྱིས་བརླབ་བོ། །འཚལ་ན་སྙས་ཕྲན་ལྱ་བར་བགྱི་སྐེ། གླེགས་བུ་བྱེད་དང་གཉིས་པར་བགྱིའོ། །བར་ཆད་མ་བྱུང་ན་བཀྲུའོ། །བརྐྱུང་ངོ་། །དྲའོ། །བསླེབ་བ། ། འ<line>
