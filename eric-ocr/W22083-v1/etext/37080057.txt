དག་གིས་འོ་མ་དང་། ཞོ་དང་། སར་སར་དང་། ཞུན་པར་དང་། ཐར་གྱི་སྲིང་ཁུ་དང་། གཞན་ཡང་ཡོ་བྱད་ཀྱིུ་བྱེ་བྲག་གསེ་གཙོ་བོ་དག་གིས་སྲིང་བར་བྱེད་།།ས<line>

གཞག་པ་དང་།གཞག་པ་དང་། འབྱེད་པ་དང་།ཀློག་པའི་ཕ་རོལ་ཏུ་སོན་ཅིང་། དེའི་ཚེན་བྲས་ཟེའི་སྤྱོད་ཚུལ་དང་།སྤོད་པ་དང་། གཙང་སྒྲ་དང་། ཀུན་ཏུ་སྤོད་པ་དང་ ས<line>
ཐལ་བ་བླང་བ་དང་། བྱ་མ་བུམ་བྤང་བ་དང་། ས་བླང་བ་དང་། ལག་སྦངས་དང་། ཐོར་རུགས་དང་། རབ་སྤགས་བྱ་བ་དང་།ཀྱེ་ཞེས་བྱ་བ་དང་། ངེས་བརྗོད་ཀྱི་རིག་བྱེད་དང་ ས<line>
སརྗོད་སྦྱིན་གྱི་རིག་བྱེད་དང་། སྐན་ཚག་གི་རིག་བྱེད་དང་། སྲིད་སྲུང་གི་རིག་བྱེད་དང་། བྲམ་ཟེའི་ལས་དྲུག་པོ་མཆོད་སྦྲིན་བྱེད་པ་དང་། མརོད་སྤིན་བྱེད་དུ་འཇུག་པ་དང་། ཀློག་པ་དང་འ<line>
ཀློག་ཏུ་འཇུག་པ་དང་། སྦྱིན་པ་དང་། ཡེན་པ་ལ་མཁས་པའི་བྲས་ཟེའི་རོག་བྱེད་དང་། རིག་བྱེད་ཀྱི་ཡན་ལག་གི་ཕ་རོལ་ཏུ་སོན་པ། ཐེ་ལྟ་བུའི་ཤེས་པས་རང་གི་སྲུབ་གསལ་བར་བྱེད་པ་ ས<line>
