རལ་ན་ལྟུང་བ་མང་དོ་། །བརྒྱ་ལྟར་ཡང་རྒྱལ་པོ་དཔུང་ག་ཚོགས་ཡན་ལག་པ་གཞག་བསྐོར་པ་གླང་པོ་ཆེ་ཚོགས་དང་། རྟ་པར་ཐ་ཚོགས་དང་གང་རྟ་པའི་<line>
ཚོགས་དང་། དཔུང་བུ་ཆུང་ཚོགས་ཅན་ཀུན་དགའ་ར་བའི་འགྱུར་ཞུགས་ནས་། སྦར་ཐབས་སུ་སྐུར་པ་སྒྲོགས་ཤིག་། ཤེས་ལྡན་ཤཱཀྱའི་བུའི་དགེ་སྦྱོང་དག་ཟུང་ཞིག་ཅེས་<line>
བྱ་བ་ནས་། གོང་ཕུལ་སྦྱོང་གི་གཟར་ཇི་སྐར་བསྟན་པ་དེ་བཞིན་དུ་རྒྱས་པར་རྫོགས་པར་བྱ་སོ་། །དེ་ན་དབྱར་བཅས་པར་དམ་བཅས་པ་དགེ་སློང་བོས་སྙམ་དུ་སེམས་ཏེ་། གནས་<line>
འབར་བདག་དབྱར་གནས་པར་དམ་བཅས་ན་། རྒྱལ་པོ་དཔུང་གཅིག་ཡན་ལག་བཞག་བསྒོ་པ་གླང་པོ་ཚོགས་དང་། རྟ་ཚོགས་ཅན་ཞེས་བྱ་བ་གོང་མ་བཀལ་ཏེ་། གལ་<line>
ཏེ་གནས་འབར་བདག་དབྱར་གནས་ན་། གཞན་བདག་སྲོག་གི་བར་ཆད་དང་། དགེ་སློང་དག་ཚུལ་རྒྱ་བར་ཆད་དང་། ཚངས་པར་སྤྱོད་པའི་བར་དུ་འགྱུར་གྱིས་། མ་ལ་ལ་<line>
བདག་གནས་འདྲ་ནས་འགྲོ་འོ་སྙམ་ནས་། དེས་དེ་ལ་བུས་བྱིན་བྱས་ཏེ་། གནས་དེ་ནས་སོང་བ་ལས་དབྱར་རལ་ན་ལྟུང་བ་མེད་དོ་། །འདི་ལྟར་སང་དགེ་སྦྱོང་གིས་སྐྱེས་པ་འམ་། བུ་ཟེ་<line>
ངམ་། མ་ནིང་གང་ལ་བརྟེན་ནས་། དབྱར་གནས་པར་དམ་བཅས་པར་གྱུར་པ་དེ་རྒྱལ་པོས་བཟུང་ངམ་། བཙོན་བཟུང་ངམ་། བསད་དམ་། དྭ་རིང་ཐོགས་སམ་། ཕུལ་བའི་<line>
