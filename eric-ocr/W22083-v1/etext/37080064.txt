གཞོན་ནུ་མ་སྐྱེས་དགྲ་། གུམ་རྒྱལ་པོ་ར་བི་ང་ཡང་འོ་ང་གཉིས་ཀྱི་དུས་སྟོན་ཉམས་སུ་མྱོང་བར་བྱེད་དུ་བཏང་བ་ཐོས་ནས་། དེས་ཀྱང་བྲམ་ཟེ་ཁྱེའུ་རྒྱལ་ལ་སྨྲས་པ་། བུ་སོང་ལ་། <line>
གྲུ་རྒྱལ་བ་བོ་དང་ཡང་འདྲ་ཉེས་ཀྱང་དུས་སྟོན་ཉམས་སུ་མྱོང་བར་རྒྱས་ལ་། དེ་ན་སྟན་བཞི་རྒྱལ་པོ་བསྟན་དང་། བདུན་ན་འདོན་གྱི་སྟན་དང་། རྒོལ་བའི་སྟན་དང་། འཇོག་<line>
ཟ་སྟན་བསམས་པ་ལས་། ཁྱོད་ཀྱིས་རྒྱལ་པོ་སྟན་དང་། ན་འདོན་གྱི་སྟན་ཞོག་ལ་། མཆོག་ཐོབ་སྟན་ལ་དབུ་གྲོ་དུང་ཚོལ་བ་ཞོག་སྟེ་། རྒོལ་བའི་སྟན་ལ་འདུག་འདུག་<line>
ཅིག་། ཇི་ཙམ་ན་ཉམ་འཆང་བ་ནས་། རྔ་ཙམ་ན་ནུབ་པར་འགྱུར་བ་དེའི་ཟེར་དུ་དུ་ཁྱེད་དང་འདྲ་བའི་རྒོལ་བ་གཞན་མེད་དོ་། །དེ་ཡང་སོང་སྟེ་རྒྱལ་པོའི་སྟན་དང་། མདུན་ཟླ་འདོན་གྱི་རྐྱེན་སྐྱེད་<line>
བཞག་པས་། མཆོག་ཐོབ་པའི་བསྙེན་ལ་བརྒྱ་གུ་དང་སེལ་བ་བཞག་སྟེ་། རྒོལ་བ་བསྟན་ལ་འདུག་གོ་། རོལ་མོའི་སྒྲ་དག་བྱུང་བ་ན་། དགའ་ཁ་དག་གསོ་ཏེ་ཏེ་ཀ་བརྒྱད་དུ་གླུ་བླངས་པ་དང་། །།་ཕྱོགས་།།་<line>
མི་དག་གིས་ཟོ་རྡུང་བ་དང་གྲུ་བླང་བར་བརྩམས་པ་ན་། བྲམ་ཟེའི་ཁྱེའུ་དེ་གནས་དབང་པོ་དག་ཕྱིར་ཟན་ཏེ་འཁོད་པ་སྐྱེ་བོ་བལ་བོ་ཆེ་སྨྲེ་ཚོགས་ཀྱིས་མཐོང་ནས་། དེ་དག་གིས་སྦྱར་པ་<line>
