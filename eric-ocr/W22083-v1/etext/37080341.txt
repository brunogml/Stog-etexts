མཐོང་ཞབས་། རྒ་ཤ་འབྱུང་བ་དང་། རྒ་ཤ་འགོག་པ་དང་། རྒ་ཤ་འགོག་པར་འདྲ་ལམ་མཛད་ངོ་། །སྐྱེ་བ་དང་། ཟབ་པ་དང་། ལམ་པ་དང་འང་<line>
སྤང་པ་དང་། ཚོར་བ་དང་། རེག་པ་དང་། སྐྱེ་བཅའ་དྲུག་དང་། ཟས་དང་གཟུགས་དང་། རྣམ་པར་ཤེས་པ་དང་འདུ་གྱུར་རྣམས་མཐོང་ཞིང་འདུ་བྱེད་རབས་<line>
འབྱུང་བ་དང་། འདུ་བྱེད་མས་འགོག་པ་དང་། འདུ་ཁྱེར་པས་འགོག་འཐོབ་བའི་ལམ་མཐོང་སྟེ་། ངས་རང་མངོན་པར་ཤེས་པས་མངོན་པར་རྫོགས་པར་བྱང་ཟབ་ནས་<line>
དགེ་སློང་པ་དང་། དགེ་དགེ་ངས་པས་རྣམས་དང་། དགེ་སློང་རྣམས་དང་། དགེ་སེར་དཔའ་རྣམས་དང་། གཞན་ཡང་སྔགས་ལྡན་དང་། དགེ་སྦྱོང་དང་། ཟ་བ་པ་དང་འབྱོར་བ་<line>
ཅན་དང་། ཀུན་རྒྱ་ཀྱང་ཚོགས་ལ་འཇོག་དོ་། །དེ་ལ་དགེ་སློང་གིས་ཀྱང་ཡང་དག་པར་བསླབ་པར་བསླབ་པར་བྱེད་པ་བརྩོན་ཞིང་། རིགས་པ་དང་ལྡན་པ་། ཆོས་དང་ཐར་པ་། དགེ་བ་<line>
བསླབ་པར་བྱེད་དོ་། །དགེ་སློང་དང་། དགེ་དགེ་བཅོམ་དང་། དགེ་སློང་མས་ཀྱང་བ་ཡང་དག་པར་བསླབ་པར་གྱུར་པ་ཞེ་། བཏགས་པ་དང་ལྡན་པ་། ཆས་དང་ཐོས་པ་། དགའ་བ་<line>
པར་བྱེད་དེ་། དེ་ལྟར་ཚངས་པར་འགྱུར་བ་འམ་རྒྱ་ཆེ་བ་། སྐྱེ་བ་མང་པོ་ལ་བམ་པ་། ཡང་དག་པར་གྱུར་པ་། ལྔ་དང་འཐོབ་བར་ཤ་ལ་ཡང་དག་པར་རབ་ཏུ་བསྟན་པ་པར་རོ་། །།་<line>
